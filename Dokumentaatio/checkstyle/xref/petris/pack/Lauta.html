22  public class Lauta extends JPanel implements ActionListener {
23  
24      private static final int LAUTA_LEVEYS = 15;
25      private static final int LAUTA_KORKEUS = 30;
26      private static final Color[] VARIT = {new Color(0, 0, 0)};
27      private Timer kello;
28      private int pudotetutPalat = 0;
29      private boolean onKaynnissa = false;
30      private boolean onPause = false;
31      private boolean palaPudonnutLoppuun = false;
32      private int curX = 0;
33      private int curY = 0;
34      private JLabel ylaReuna;
35      private Pala palaPelissa;
36      private Pentomino[] lauta;
37  
38      /**
39       * Lauta-luokan konstruktori, joka myös käynnistää pelin.
40       * @param parent
41       */
42      public Lauta(Petris parent) {
43          setFocusable(true);
44          palaPelissa = new Pala();
45          kello = new Timer(400, this);
46          kello.start();
47          ylaReuna = parent.getYlaReuna();
48          lauta = new Pentomino[LAUTA_LEVEYS * LAUTA_KORKEUS];
49          tyhjennaLauta(); // konstruktori tekee tyhjän laudan.
50          this.pudotetutPalat = 0; // voi aiheuttaa tuloksen nollautumisen
51      }
52  
53      /**
54       * Käsittelee pelin saaman keyboard syötteen, sisältää myös hieman pelin
55       * logiikkaa.
56       * @param ae
57       */
58      @Override
59      public void actionPerformed(ActionEvent ae) {
60          // jos pala pudonnut loppuun eli true, tehdään uusi pala, muuten pudotetaan lisää.
61          if (palaPudonnutLoppuun) {
62              palaPudonnutLoppuun = false; // alustetaan uutta palaa varten
63              uusiPala();
64          } else {
65              yksiAskelAlaspain();
66          }
67  
68      }
69  
70      /**
71       * Palauttaa Petrispelin laudan leveyden Integerinä.
72       * @return
73       */
74      public int laudanLeveys() {
75          return (int) getSize().getWidth() / LAUTA_LEVEYS;
76      }
77  
78      /**
79       * Palauttaa Petrispelin laudan korkeuden Integerinä.
80       * @return
81       */
82      public int laudanKorkeus() {
83          return (int) getSize().getHeight() / LAUTA_KORKEUS;
84      }
85  
86      /**
87       * Tyhjentää Petrispelin laudan asettamalla kaikkien koordinaattien
88       * Pentominoksi muodottoman eli tyhjän.
89       */
90      public void tyhjennaLauta() {
91          for (int i = 0; i < LAUTA_LEVEYS * LAUTA_KORKEUS; i++) {
92              lauta[i] = Pentomino.Muodoton;
93          }
94      }
95  
96      /**
97       * Palauttaa Pentominon kohdassa (x,y), "Muodoton" tarkoitaa
98       * käytännössä tyhjää.
99       * @param x
100      * @param y
101      * @return
102      */
103     public Pentomino palaKohdassa(int x, int y) {
104         return lauta[y * LAUTA_LEVEYS + x];
105     }
106 
107     /**
108      *
109      */
110     public void palanPudotus() {
111         for (int i = 0; i < 5; i++) {
112             int x = curX + palaPelissa.getX(i);
113             int y = curY - palaPelissa.getY(i);
114         }
115         if (!palaPudonnutLoppuun) {
116             uusiPala();
117         }
118     }
119 
120     // palan liikuttamista varten myös
121 
122     /**
123      * Luo uuden Palan peliin, sisältää myös samalla pelin logiikkaa.
124      */
125     public void uusiPala() {
126         palaPelissa.setRandom();
127         curX = LAUTA_LEVEYS / 2 + 1;
128         curY = LAUTA_KORKEUS - 1 + palaPelissa.minY();
129 
130         // ei enää tilaa uudelle palalle
131         if (!yritaLiikkua(palaPelissa, curX, curY - 1)) {
132             palaPelissa.setPala(Pentomino.Muodoton);
133             kello.stop();
134             onKaynnissa = false;
135             ylaReuna.setText("Peli loppui!");
136 
137         }
138 
139     }
140 
141     /**
142      * Kutsuu toista metodia, käytännössä liikuttaa palaa askeleen alaspäin.
143      */
144     public void yksiAskelAlaspain() {
145         palanPudotus();
146     }
147

162     public void asetaPelissaOlevanPalanMuoto(Pentomino pento) {
163         palaPelissa.setMuoto(pento);
164     }

166     /**
167      * Piirtää lautaa ja Paloja laudalle.
168      * @param g
169      * @param x
170      * @param y
171      * @param muoto
172      */
173     public void piirra(Graphics g, int x, int y, Pentomino muoto) {
174         Color vari = muoto.color;
175         g.setColor(vari);
176         g.fillRect(x + 1, y + 1, laudanLeveys() - 2, laudanKorkeus() - 2);
177         g.setColor(vari.brighter());
178         g.drawLine(x, y, x + laudanLeveys() - 1, y);
179         g.drawLine(x, y + laudanKorkeus() - 1, x, y);
180         g.setColor(vari.darker());
181         g.drawLine(x + 1, y + laudanKorkeus() - 1, x + laudanLeveys(), y - laudanKorkeus() - 1);
182         g.drawLine(x + laudanLeveys() - 1, y + laudanKorkeus() - 1, x + laudanLeveys() - 1, y + 1);
183     }
184 
185     /**
186      * Piirtää laudan kokonaisuutena.
187      * @param g
188      */
189     public void piirraLauta(Graphics g) {
190         super.paint(g);
191         Dimension koko = getSize();
192 
193         int laudanKatto = (int) koko.getHeight() - LAUTA_KORKEUS * laudanKorkeus();
194 
195         for (int i = 0; i < LAUTA_KORKEUS; i++) {
196             for (int j = 0; j < LAUTA_LEVEYS; j++) {
197                 Pentomino pentomino = palaKohdassa(j, LAUTA_KORKEUS - i - 1);
198 
199                 if (pentomino != Pentomino.Muodoton) {
200                     piirra(g, j * laudanLeveys(), laudanKatto + i * laudanKorkeus(), pentomino);
201                 }
202             }
203         }
204     }
205 
206     private boolean yritaLiikkua(Pala uusiPala, int uusiX, int uusiY) {
207         for (int i = 0; i < 5; i++) {
208             int x = uusiX + uusiPala.getX(i);
209             int y = uusiY + uusiPala.getY(i);
210 
211             // Liikkuminen yli laudan
212             if (x < 0 || x >= LAUTA_LEVEYS || y < 0 || y >= LAUTA_KORKEUS) {
213                 return false;
214             }
215             // yrittää siirtyä toisen palan päälle
216             if (palaKohdassa(x, y) != Pentomino.Muodoton) {
217                 return false;
218             }
219             palaPelissa = uusiPala;
220             curX = uusiX;
221             curY = uusiY;
222             repaint();
223         }
224 
225         return true;
226     }
